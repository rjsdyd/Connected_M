# 프론트 명세 내용
- user entity 컬럼 추가 : 
- content entity 컬럼 추가 : view_count, wish_count
- UserReview entity status 역할 추가 변경
- User와 Content 사이 중간 브릿지 테이블 : Wishlist 관계 매핑 최적화
- LoginLog 엔티티 추가 : 유저기록 관리

## ERD
```sql
CREATE TABLE `ChatSession` (
	`id`	BIGINT	NOT NULL	COMMENT '대화방 식별자',
	`session_title`	VARCHAR(255)	NULL	COMMENT '세션별 주',
	`user_id`	BIGINT	NOT NULL
);

CREATE TABLE `ChatMessage` (
	`id`	BIGINT	NOT NULL	COMMENT '메시지 식별자',
	`role`	VARCHAR(20)	NOT NULL	COMMENT 'user/assistant',
	`message`	TEXT	NOT NULL	COMMENT '대화 본문',
	`session_id`	BIGINT	NOT NULL	COMMENT '대화방 식별자'
);

CREATE TABLE `AnalysisCache` (
	`id`	BIGINT	NOT NULL	COMMENT '분석 결과 식별자',
	`summary`	TEXT	NOT NULL	COMMENT 'AI 생성 줄거리 요약',
	`positive_ratio`	DECIMAL(3,1)	NOT NULL	COMMENT '예 : 8.7',
	`top_keywords`	VARCHAR(255)	NULL	COMMENT '주요 키워드',
	`content_id`	BIGINT	NOT NULL	COMMENT '시스템 관리용 PK',
	`embedding_vector`	LONGTEXT	NULL	COMMENT '문장 의미 데이터(AI가 생성한 숫자 뭉치)',
	`search_keywords`	VARCHAR(500)	NULL	COMMENT '검색 가중치용 태그(AI키워드 +  분위기 태그)'
);

CREATE TABLE `Wishlist` (
	`id`	BIGINT	NOT NULL	COMMENT '위시리스트 식별자',
	`created_at`	DATETIME	NOT NULL	DEFAULT NOW()	COMMENT '찜한 시간',
	`content_id`	BIGINT	NOT NULL	COMMENT '시스템 관리용 PK',
	`user_id`	BIGINT	NOT NULL
);

CREATE TABLE `RecentView` (
	`id`	BIGINT	NOT NULL,
	`user_id`	BIGINT	NOT NULL,
	`content_id`	BIGINT	NOT NULL	COMMENT '시스템 관리용 PK',
	`viewed_at`	DATETIME	NOT NULL
);

CREATE TABLE `Content` (
	`id`	BIGINT	NOT NULL	COMMENT '시스템 관리용 PK',
	`tmdb_id`	VARCHAR(50)	NOT NULL	COMMENT 'TMDB API 식별자',
	`cine21_id`	VARCHAR(50)	NULL,
	`title`	VARCHAR(255)	NOT NULL	COMMENT '작품 제목',
	`overview`	TEXT	NULL	COMMENT '긴 내용 수용 가능',
	`poster_path`	VARCHAR(255)	NULL	COMMENT '이미지 경로',
	`ott_logos`	VARCHAR(255)	NULL	COMMENT '넷플릭스 등 아이콘',
	`backdrop_path`	VARCHAR(200)	NULL,
	`runtime`	INTEGER	NULL,
	`age_rating`	VARCHAR(50)	NULL,
	`view_count`	INT	NULL	COMMENT '누적 조회수',
	`wish_count`	INT	NULL	COMMENT '찜 누적수'
);

CREATE TABLE `UserReview` (
	`id`	BIGINT	NOT NULL	COMMENT 'pk',
	`rating`	VARCHAR(10)	NOT NULL	COMMENT '사용자 평점',
	`comment`	TEXT	NOT NULL	COMMENT '사용자 리뷰 본문',
	`created_at`	DATETIME	NOT NULL	DEFAULT NOW()	COMMENT '작성일자',
	`user_id`	BIGINT	NOT NULL	COMMENT 'fk',
	`content_id`	BIGINT	NOT NULL	COMMENT 'fk',
	`report_count`	INT	NOT NULL	DEFAULT 0,
	`status`	VARCHAR(20)	NOT NULL	DEFAULT 'NORMAL'	COMMENT '리뷰상태 (NORMAL, HIDDEN)'
);

CREATE TABLE `ContentGenre` (
	`content_id`	BIGINT	NOT NULL	COMMENT '시스템 관리용 PK',
	`genre_id`	BIGINT	NOT NULL	COMMENT '장르 고유 번호'
);

CREATE TABLE `ExpertReview` (
	`id`	BIGINT	NOT NULL	COMMENT '리뷰 고유 번호',
	`content_id`	BIGINT	NOT NULL,
	`analysis_id`	BIGINT	NOT NULL,
	`critic_name`	VARCHAR(50)	NOT NULL	COMMENT '평론가이름',
	`rating`	VARCHAR(10)	NULL,
	`comment`	TEXT	NOT NULL	COMMENT '사용자 리뷰',
	`source`	VARCHAR(50)	NULL	DEFAULT 'Cine21'	COMMENT '출처',
	`source_id`	BIGINT	NOT NULL
);

CREATE TABLE `Cine21Source` (
	`id`	BIGINT	NOT NULL,
	`external_movie_id`	VARCHAR(50)	NOT NULL	DEFAULT Unique	COMMENT '씨네21 고유 영화 번호',
	`raw_title`	VARCHAR(250)	NOT NULL	COMMENT '크롤링 시점의 원본 제목 보존용',
	`crawl_date`	DATETIME	NOT NULL
);

CREATE TABLE `Genre` (
	`id`	BIGINT	NOT NULL	COMMENT '장르 고유 번호',
	`name`	VARCHAR(50)	NOT NULL	COMMENT '액션, SF 등'
);

CREATE TABLE `User` (
	`id`	BIGINT	NOT NULL,
	`email`	VARCHAR(100)	NOT NULL	DEFAULT UK	COMMENT '로그인용(중복불가)',
	`password`	VARCHAR(255)	NOT NULL	COMMENT '암호화된 비밀번호',
	`realName`	VARCHAR(50)	NOT NULL	COMMENT '사용자 이름',
	`phoneNumber`	VARCHAR(20)	NOT NULL,
	`nickname`	VARCHAR(50)	NOT NULL	COMMENT '서비스용 이름',
	`role`	VARCHAR(20)	NOT NULL	DEFAULT 'ROLE_USER'	COMMENT '사용자 권한(ROLE_USER, ROLE_ADMIN)',
	`provider`	VARCHAR(20)	NOT NULL	COMMENT 'Enum(LOCAL, KAKAO, GOOGLE)',
	`provider_id`	VARCHAR(255)	NOT NULL,
	`password_reset_token`	VARCHAR(255)	NULL	COMMENT 'unique',
	`password_reset_expiry`	DATETIME	NULL,
	`status`	VARCHAR(20)	NOT NULL	DEFAULT 'ACTIVE'	COMMENT '사용자 상태 (ACTIVE, BANNED, PENDING)',
	`reported_count`	INT	NOT NULL	DEFAULT 0,
	`created_at`	DATETIME	NULL	DEFAULT NOW()	COMMENT '가입일시',
	`last_login_at`	DATETIME	NULL
);

CREATE TABLE `login_log` (
	`id`	BIGINT	NOT NULL,
	`user_id`	BIGINT	NOT NULL	COMMENT '사용자아이디 참조',
	`login_at`	DATETIME	NULL,
	`ip_address`	VARCHAR(45)	NULL	COMMENT 'IPv6대응',
	`device_info`	VARCHAR(255)	NULL
);

ALTER TABLE `ChatSession` ADD CONSTRAINT `PK_CHATSESSION` PRIMARY KEY (
	`id`
);

ALTER TABLE `ChatMessage` ADD CONSTRAINT `PK_CHATMESSAGE` PRIMARY KEY (
	`id`
);

ALTER TABLE `AnalysisCache` ADD CONSTRAINT `PK_ANALYSISCACHE` PRIMARY KEY (
	`id`
);

ALTER TABLE `Wishlist` ADD CONSTRAINT `PK_WISHLIST` PRIMARY KEY (
	`id`
);

ALTER TABLE `RecentView` ADD CONSTRAINT `PK_RECENTVIEW` PRIMARY KEY (
	`id`
);

ALTER TABLE `Content` ADD CONSTRAINT `PK_CONTENT` PRIMARY KEY (
	`id`
);

ALTER TABLE `UserReview` ADD CONSTRAINT `PK_USERREVIEW` PRIMARY KEY (
	`id`
);

ALTER TABLE `ContentGenre` ADD CONSTRAINT `PK_CONTENTGENRE` PRIMARY KEY (
	`content_id`,
	`genre_id`
);

ALTER TABLE `ExpertReview` ADD CONSTRAINT `PK_EXPERTREVIEW` PRIMARY KEY (
	`id`
);

ALTER TABLE `Cine21Source` ADD CONSTRAINT `PK_CINE21SOURCE` PRIMARY KEY (
	`id`
);

ALTER TABLE `Genre` ADD CONSTRAINT `PK_GENRE` PRIMARY KEY (
	`id`
);

ALTER TABLE `User` ADD CONSTRAINT `PK_USER` PRIMARY KEY (
	`id`
);

ALTER TABLE `login_log` ADD CONSTRAINT `PK_LOGIN_LOG` PRIMARY KEY (
	`id`
);

ALTER TABLE `ContentGenre` ADD CONSTRAINT `FK_Content_TO_ContentGenre_1` FOREIGN KEY (
	`content_id`
)
REFERENCES `Content` (
	`id`
);

ALTER TABLE `ContentGenre` ADD CONSTRAINT `FK_Genre_TO_ContentGenre_1` FOREIGN KEY (
	`genre_id`
)
REFERENCES `Genre` (
	`id`
);
```
